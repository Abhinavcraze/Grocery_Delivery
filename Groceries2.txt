document.addEventListener('DOMContentLoaded', () => {
    // 1. Get Price
    const savedAmount = localStorage.getItem('orderTotal');
    const displayElement = document.querySelector('.font-black.text-lg');
    if (savedAmount && displayElement) displayElement.innerText = savedAmount;

    // 2. Get Slot
    const savedSlot = localStorage.getItem('deliverySlot');
    const slotDisplay = document.getElementById('display-slot');
    if (savedSlot && slotDisplay) {
        slotDisplay.innerText = savedSlot;
        
        // If it's not "Instant", stop the timer simulation
        if (savedSlot !== "Instant (10-15 Mins)") {
            document.getElementById('timer').innerText = "SCHEDULED";
        }
    }
});
